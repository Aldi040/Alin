##C. Sifat-Sifat Determinan

#Sifat-Sifat determinan
print("Sifat-Sifat determinan")
print("determinan Matriks")
# Matriks identitas memiliki determinan 1.
print("Determinan A (matriks identitas):", int(detA))
print("Determinan B (matriks identitas):", int(detB))
# matriks C
print("Determinan C matriks C: ",int(detC))
# matriks c dengan merubah baris akan mengubah tanda dari determinan
print("Determinan D (matriks c dengan merubah baris akan mengubah tanda dari determinan):",int(detD))
# Jika suatu baris matriks adalah kelipatan dari baris lainnya, determinannya akan menjadi nol.
print("Determinan E (baris 2 merupakan kelipatan dari baris 3):", int(detE))

# Jika suatu kolom matriks adalah kelipatan dari kolom lainnya, determinannya juga akan menjadi nol.
print("Determinan F (kolom 3 merupakan kelipatan dari kolom 1):", int(detF))
print("Determinan G (kolom 3 merupakan kelipatan dari kolom 2):", int(detG))

# Untuk matriks yang merupakan kelipatan dari matriks lainnya, determinannya adalah kelipatan pangkat ordo dari matriks tersebut, dikalikan dengan determinan matriks kelipatannya.
print("Determinan H (matriks H merupakan kelipatan dari matriks C):", int(detH))
# Jika suatu matriks adalah hasil penjumlahan dari dua matriks, determinannya adalah hasil penjumlahan determinan matriks-matriks tersebut.
print("Determinan M (hasil penjumlahan matriks K dan L):", detM)

# Jika suatu matriks membentuk segitiga atas dengan nol di atas atau di bawah diagonal, determinannya adalah hasil perkalian diagonalnya.
print("Determinan N (matriks N membentuk segitiga atas):", detN)

# Jika dua matriks dikalikan, determinan dari hasil matriks adalah hasil perkalian determinan matriks tersebut.
print("Determinan P (hasil perkalian matriks O dan Q):", detP)
print("Determinan Q (matriks Q):", detQ)
print("determinan T:",detT)
